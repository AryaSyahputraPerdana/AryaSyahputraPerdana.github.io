📊 1. Exploratory Data Analysis (EDA)

Membaca dataset berukuran besar (466k+ baris) dan memvisualisasikan korelasi antar fitur numerik menggunakan fungsi heatmap dari library Seaborn.

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load dataset
file_path = "loan_data_2007_2014.csv"
df = pd.read_csv(file_path)

# Cek korelasi antar fitur numerik
numeric_df = df.select_dtypes(include=['number'])
plt.figure(figsize=(10, 8))
sns.heatmap(numeric_df.corr(), cmap='coolwarm', annot=False, linewidths=0.5)

plt.title("Heatmap Korelasi Antar Fitur Numerik")
plt.show()

🧹 2. Data Cleaning & Pre-processing

Menangani nilai kosong (Missing Values) menggunakan median untuk angka dan modus untuk teks.

# Isi kolom numerik dengan median
num_cols = df.select_dtypes(include=['float64', 'int64']).columns
df[num_cols] = df[num_cols].apply(lambda x: x.fillna(x.median()))

# Isi kolom kategorikal dengan modus (nilai yang paling sering muncul)
cat_cols = df.select_dtypes(include=['object']).columns
df[cat_cols] = df[cat_cols].apply(lambda x: x.fillna(x.mode()[0]))

print("\nMissing values setelah handling:")
print(df.isnull().sum())

⚙️ 3. Feature Encoding & Standard Scaling

Mengubah data teks menjadi angka (Label Encoding) dan menormalisasi rentang nilai numerik. Objek scaler dan encoder disimpan (joblib) untuk kebutuhan inferensi selanjutnya.

from sklearn.preprocessing import StandardScaler, LabelEncoder
import joblib

# Proses Label encoding
label_encoders = {} 
for col in categorical_cols:
    le = LabelEncoder()
    unique_values = df[col].unique().tolist()
    if "Unknown" not in unique_values:
        unique_values.append("Unknown")
    
    le.fit(unique_values)
    df[col] = le.transform(df[col])
    label_encoders[col] = le

joblib.dump(label_encoders, "label_encoders.pkl")

# Inisialisasi StandardScaler
scaler = StandardScaler()
df[numerical_cols] = scaler.fit_transform(df[numerical_cols].copy())

joblib.dump(scaler, "scaler.pkl")

def evaluate_model(model, X_test, y_test, model_name):
    y_pred = model.predict(X_test)
    y_pred_proba = model.predict_proba(X_test)
    
    accuracy = accuracy_score(y_test, y_pred)
    roc_auc = roc_auc_score(y_test, y_pred_proba, multi_class='ovr')
    
    print(f"\nEvaluasi {model_name}")
    print(f"Accuracy: {accuracy:.4f}")
    print(f"ROC AUC Score: {roc_auc:.4f}")
